cmake_minimum_version(VERSION 3.0)

project(static_engine)

file(GLOB_RECURSE sources engine/src/* engine/include/*)
file(GLOB_RECURSE sources_test engine/test/*)
file(GLOB_RECURSE data engine/resources/*)

add_executable(static_eng ${sources} ${data})

target_compile_options(static_eng PUBLIC -std=c++23 -Wall -Wfloat-conversion)

target_include_directories(static_eng PUBLIC engine/src engine/include)

file(COPY ${data} DESTINATION resources)

# Dependencies

find_package(Vulkan REQUIRED)

target_link_libraries(static_eng PRIVATE Vulkan::Vulkan)

find_package(GTest)

if (GTEST_FOUND)
    add_executable(unit_tests ${sources_test} ${sources})

    target_compile_definitions(unit_tests PUBLIC UNIT_TESTS)

    # Use our executable as a link library so we inherit compiler options and dependencies
    set_target_properties(static_eng PROPERTIES ENABLE_EXPORTS on)

    target_link_libraries(unit_tests PUBLIC 
        ${GTEST_BOTH_LIBRARIES}
        static_eng
    )

    target_include_directories(unit_tests PUBLIC
        ${GTEST_INCLUDE_DIRS}
    )
endif()

install(TARGETS static_eng DESTINATION engine)
install(DIRECTORY resources DESTINATION engine)

set(CPACK_PACKAGE_NAME "MyExample")
set(CPACK_PACKAGE_VERSION "1.0.0")

set(CPACK_MONOLITHIC_INSTALL 1)

include(CPack)
